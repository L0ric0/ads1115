66  int16_t ADS1115::read() const
67  {
68  if (m_config.mode == MODE::SINGLE_CONV) {
69  // to start a conversion in single conversion mode the MSB of the config
70  // register has to be set
71  write_word(conf_reg_addr, m_config.to_bytes() | 0x8000);
72  // when the conversion starts the ADS1115 resets the MSB and sets it when the conversion
73  // is done
74  while (!(read_word(conf_reg_addr) & 0x8000)) {
75  // poll every millisecond, as the shortest conversion the device can do takes 1.2ms
76  std::this_thread::sleep_for(std::chrono::milliseconds(1));
77  }
78  }
79 
80  uint16_t data = read_word(conv_reg_addr);
